js之Object与Function的三种属性(__proto__、prototype和constructor)之间的关系

2016-04-01

// true true true true true true true true true true true true
console.log(
    //Object && Function
    Object.constructor === Function,
    Object.__proto__.constructor === Function,
    //Object && A
    Object.__proto__ === Object.constructor.__proto__,
    Object.__proto__ === Object.constructor.prototype,
    //Object && B
    Object.prototype === Object.__proto__.__proto__,
    Object.prototype.constructor === Object,
    //Object && null
    Object.prototype.__proto__ === null,
    //Function && A
    Function.prototype === Object.__proto__,
    Function.__proto__ === Object.__proto__,
    Object.__proto__.constructor === Function,
    //A && B
    Object.__proto__.__proto__ === Object.prototype,
    //long way
    Object.constructor.prototype.__proto__.constructor === Object
);

    Object.__proto__ === Object.constructor.prototype, Function.__proto__ === Function.constructor.prototype，理解__proto__属性为构造函数的原型的引用。
